CREATE DATABASE eshopengeto /*!40100 DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci */ /*!80016 DEFAULT ENCRYPTION='N' */;
CREATE DATABASE todolist /*!40100 DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci */ /*!80016 DEFAULT ENCRYPTION='N' */;

CREATE TABLE engetoproductslist (
  id BIGINT(20) NOT NULL AUTO_INCREMENT,
  partNo int NOT NULL,
  name varchar(45) COLLATE utf8_czech_ci NOT NULL,
  description varchar(255) COLLATE utf8_czech_ci DEFAULT NULL,
  isForSale tinyint NOT NULL,
  price double NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY partNo_UNIQUE (partNo)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb3 COLLATE=utf8_czech_ci

CREATE TABLE todoitem (
  id bigint NOT NULL AUTO_INCREMENT,
  text varchar(255) COLLATE utf8_czech_ci NOT NULL,
  isCompleted tinyint NOT NULL,
  PRIMARY KEY (id)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb3 COLLATE=utf8_czech_ci
